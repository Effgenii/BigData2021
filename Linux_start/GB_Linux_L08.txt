4. * Используя Dockerfile, собрать связку nginx + PHP-FPM в одном контейнере.

FROM ubuntu
MAINTAINER Student
RUN apt-get update
RUN apt-get install nginx -y php php-fpm php-mysqli
VOLUME "/var/www/html"
EXPOSE 80
CMD /usr/sbin/nginx -g "daemon off;"


docker build -t main_image_nginx_PHP
