4. Создать пользователя, обладающего возможностью выполнять действия от имени суперпользователя.
-- первый вариант - создаем нового суперпользователя
student@Ubuntu-MySQL-VirtualBox:~/lesson$ sudo useradd -ou 0 -g 0 admin
student@Ubuntu-MySQL-VirtualBox:~/lesson$ sudo passwd admin
-- второй вариант - добавляем пользователя в группу sudo
usermod -aG sudo admin
-- третий вариант - добаляем в корневую группу
useradd -m -G root admin
